# Comando para verificar e configurar Database IDs na VPS
# Execute este comando na VPS:

cd /var/www/founder-dashboard && echo "=== DATABASE IDs ATUAIS ===" && grep "^NOTION_DB_" .env.local && echo "" && echo "=== VERIFICANDO SE ESTÃO CONFIGURADOS ===" && [ -z "$(grep "^NOTION_DB_KPIS=" .env.local)" ] && echo "❌ NOTION_DB_KPIS não encontrado" || echo "✅ NOTION_DB_KPIS configurado: $(grep "^NOTION_DB_KPIS=" .env.local | cut -d'=' -f2)" && [ -z "$(grep "^NOTION_DB_GOALS=" .env.local)" ] && echo "❌ NOTION_DB_GOALS não encontrado" || echo "✅ NOTION_DB_GOALS configurado: $(grep "^NOTION_DB_GOALS=" .env.local | cut -d'=' -f2)" && echo "" && echo "=== TESTE COM OS IDs ATUAIS ===" && TOKEN=$(grep "^NOTION_TOKEN=" .env.local | cut -d'=' -f2) && DB_KPIS=$(grep "^NOTION_DB_KPIS=" .env.local | cut -d'=' -f2) && if [ ! -z "$DB_KPIS" ]; then echo "Testando Database ID: $DB_KPIS" && curl -s -X POST "https://api.notion.com/v1/databases/$DB_KPIS/query" -H "Notion-Version: 2022-06-28" -H "Authorization: Bearer $TOKEN" -H "Content-Type: application/json" -d '{}' | head -c 500 && echo ""; else echo "❌ Database ID não encontrado"; fi
